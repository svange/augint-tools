510) -> StepResult:
511    from augint_tools.ide.bookmarks import (
512        bookmarks_already_set,
513        build_bookmarks_xml,
514        discover_bookmarks,
515        format_bookmark_table,
516        inject_bookmarks,
517    )
518
519    name = "bookmarks"
520
521    slots = discover_bookmarks(project_dir)
522    if not slots:
523        return _skipped(name, "No bookmarkable files found in project")
524
525    table = format_bookmark_table(slots)
526    if product_workspace_path is None:
527        result = _action_required(
528            name,
529            f"Found {len(slots)} files to bookmark but cannot locate product workspace file",
530            missing_inputs=["product_workspace_file"],
531            next_action=(
532                "Open the project in IntelliJ IDEA once, close it, then re-run. "
533                "This creates the workspace file that bookmarks are stored in."
534            ),
535        )
536        result.details = {
537            "table": table,
538            "bookmarks": [{"mnemonic": s.mnemonic, "file": s.rel} for s in slots],
539        }
540        return result
541
542    if bookmarks_already_set(product_workspace_path, slots, project_dir):
543        return _skipped(
544            name,
545            f"{len(slots)} mnemonic bookmarks already configured",
546            bookmarks=[{"mnemonic": s.mnemonic, "file": s.rel} for s in slots],
547        )
548
549    component = build_bookmarks_xml(slots, project_dir, group_name=project_name)
550    inject_result = inject_bookmarks(product_workspace_path, component, dry_run)
551
552    if inject_result.get("action") == "error":
553        return _error(name, inject_result.get("reason", "Failed to write bookmarks"))
554
555    return _ok(
556        name,
557        f"{len(slots)} mnemonic bookmarks {'would be ' if dry_run else ''}set",
558        bookmarks=[{"mnemonic": s.mnemonic, "file": s.rel} for s in slots],
559        table=table,
560        workspace_file=product_workspace_path,
561        action=inject_result["action"],
562    )

164def step_module_sdk(iml_path: str | None, sdk_name: str, dry_run: bool = False) -> StepResult:
165    name = "module_sdk"
166    if iml_path is None:
167        return _error(name, "No .iml file found")
168
169    tree, root = read_xml(iml_path)
170    if tree is None or root is None:
171        return _error(name, f"Could not read {iml_path}")
172
173    comp = root.find('.//component[@name="NewModuleRootManager"]')
174    if comp is None:
175        return _error(name, f"NewModuleRootManager not found in {iml_path}")
176
177    # Check for explicit SDK assignment (type="jdk")
178    existing_jdk = comp.find('.//orderEntry[@type="jdk"]')
179    if existing_jdk is not None:
180        if existing_jdk.get("jdkName") == sdk_name and existing_jdk.get("jdkType") == "Python SDK":
181            return _skipped(name, f"Module SDK already set to '{sdk_name}'")
182        existing_jdk.set("jdkName", sdk_name)
183        existing_jdk.set("jdkType", "Python SDK")
184        write_xml(tree, iml_path, dry_run)
185        return _ok(name, f"Module SDK set to '{sdk_name}'", sdk_name=sdk_name)
186
187    # Check for inherited SDK (type="inheritedJdk") — module uses the project SDK.
188    # This is the default IDEA layout and is correct when the project SDK is set.
189    inherited = comp.find('.//orderEntry[@type="inheritedJdk"]')
190    if inherited is not None:
191        return _skipped(
192            name,
193            f"Module inherits project SDK (set via misc.xml to '{sdk_name}')",
194            inherited=True,
195        )
196
197    # No SDK entry at all — add one
198    ET.SubElement(comp, "orderEntry", type="jdk", jdkName=sdk_name, jdkType="Python SDK")
199    write_xml(tree, iml_path, dry_run)
200    return _ok(name, f"Module SDK set to '{sdk_name}'", sdk_name=sdk_name)

def step_project_structure( iml_path: str | None, project_dir: str, project_name: str, dry_run: bool = False) -> StepResult: 
208def step_project_structure(
209    iml_path: str | None,
210    project_dir: str,
211    project_name: str,
212    dry_run: bool = False,
213) -> StepResult:
214    name = "structure"
215    if iml_path is None:
216        return _error(name, "No .iml file found")
217
218    tree, root = read_xml(iml_path)
219    if tree is None or root is None:
220        return _error(name, f"Could not read {iml_path}")
221
222    comp = root.find('.//component[@name="NewModuleRootManager"]')
223    if comp is None:
224        return _error(name, "NewModuleRootManager not found")
225
226    content = comp.find('content[@url="file://$MODULE_DIR$"]')
227    if content is None:
228        return _error(name, "<content> element not found in .iml")
229
230    def _url(rel: str) -> str:
231        return f"file://$MODULE_DIR$/{rel}"
232
233    def _has_source(rel: str) -> bool:
234        return content.find(f'sourceFolder[@url="{_url(rel)}"]') is not None
235
236    def _has_exclude(rel: str) -> bool:
237        return content.find(f'excludeFolder[@url="{_url(rel)}"]') is not None
238
239    added_sources: list[str] = []
240    added_tests: list[str] = []
241    added_excludes: list[str] = []
242
243    source_candidates = ["src", "scripts"]
244    pkg_init = os.path.join(project_dir, project_name, "__init__.py")
245    if os.path.isdir(os.path.join(project_dir, project_name)) and os.path.exists(pkg_init):
246        source_candidates.insert(0, project_name)
247
248    for d in source_candidates:
249        if not os.path.isdir(os.path.join(project_dir, d)):
250            continue
251        if not _has_source(d):
252            added_sources.append(d)
253
254    for d in ["tests", "test"]:
255        if not os.path.isdir(os.path.join(project_dir, d)):
256            continue
257        if not _has_source(d):
258            added_tests.append(d)
259
260    exclude_candidates = list(ALWAYS_EXCLUDE)
261    for p in glob.glob(os.path.join(project_dir, "*.egg-info")):
262        if os.path.isdir(p):
263            exclude_candidates.append(os.path.basename(p))
264
265    for d in exclude_candidates:
266        if not _has_exclude(d):
267            added_excludes.append(d)
268
269    if not added_sources and not added_tests and not added_excludes:
270        return _skipped(name, "Project structure already configured")
271
272    existing_excludes = content.findall("excludeFolder")
273    insert_idx = list(content).index(existing_excludes[0]) if existing_excludes else len(content)
274
275    offset = 0
276    for d in added_sources:
277        el = ET.Element("sourceFolder", url=_url(d), isTestSource="false")
278        content.insert(insert_idx + offset, el)
279        offset += 1
280
281    for d in added_tests:
282        el = ET.Element("sourceFolder", url=_url(d), isTestSource="true")
283        content.insert(insert_idx + offset, el)
284        offset += 1
285
286    for d in added_excludes:
287        ET.SubElement(content, "excludeFolder", url=_url(d))
288
289    write_xml(tree, iml_path, dry_run)
290    msg = (
291        f"added {len(added_sources)} source, {len(added_tests)} test, "
292        f"{len(added_excludes)} excluded roots"
293    )
294    return _ok(
295        name,
296        msg,
297        added_sources=added_sources,
298        added_tests=added_tests,
299        added_excludes=added_excludes,
300    )

def extract_repo_slug(remote_url: str) -> str | None: 
 87def extract_repo_slug(remote_url: str) -> str | None:
 88    """Extract owner/repo from a git remote URL.
 89
 90    Handles HTTPS, SSH, and proxy formats:
 91        https://github.com/owner/repo.git -> owner/repo
 92        git@github.com:owner/repo.git -> owner/repo
 93        http://local_proxy@127.0.0.1:8080/git/owner/repo -> owner/repo
 94    """
 95    url = remote_url.rstrip("/").removesuffix(".git")
 96
 97    # Proxy format: http://local_proxy@127.0.0.1:PORT/git/owner/repo
 98    if "local_proxy@127.0.0.1" in url and "/git/" in url:
 99        after_git = url.split("/git/", 1)[1]
100        parts = after_git.split("/")
101        if len(parts) >= 2:
102            return f"{parts[0]}/{parts[1]}"
103        return None
104
105    if "github.com" not in url:
106        return None
107
108    if url.startswith("git@"):
109        # git@github.com:owner/repo
110        parts = url.split(":")
111        if len(parts) == 2:
112            return parts[1]
113    else:
114        # https://github.com/owner/repo
115        parts = url.split("/")
116        if len(parts) >= 2:
117            return f"{parts[-2]}/{parts[-1]}"
118
119    return None

def detect_base_branch(path: pathlib.Path | None = None) -> str: 
122def detect_base_branch(path: Path | None = None) -> str:
123    """
124    Detect the base branch for the repository.
125
126    Checks for main, master, dev, develop in that order.
127
128    Args:
129        path: Repository path (defaults to current directory)
130
131    Returns:
132        Base branch name (defaults to "main" if none found)
133    """
134    candidates = ["main", "master", "dev", "develop"]
135
136    try:
137        # Get all branches
138        result = run_git(["branch", "-a"], cwd=path, check=False)
139        if result.returncode != 0:
140            return "main"
141
142        branches = [line.strip().lstrip("* ") for line in result.stdout.split("\n")]
143
144        # Check for local branches first
145        for candidate in candidates:
146            if candidate in branches:
147                return candidate
148
149        # Check for remote branches
150        for candidate in candidates:
151            if f"remotes/origin/{candidate}" in branches:
152                return candidate
153
154        return "main"
155    except Exception:
156        return "main"

25    Args:

def get_ahead_behind( path: pathlib.Path | None = None, remote_branch: str | None = None) -> tuple[int, int]: 
46def get_ahead_behind(path: Path | None = None, remote_branch: str | None = None) -> tuple[int, int]:
47    """
48    Get ahead/behind count relative to remote branch.
49
50    Args:
51        path: Repository path (defaults to current directory)
52        remote_branch: Remote branch to compare (e.g., "origin/main")
53                      If None, uses tracking branch
54
55    Returns:
56        Tuple of (ahead, behind) counts
57    """
58    try:
59        # If no remote branch specified, try to get tracking branch
60        if remote_branch is None:
61            result = run_git(
62                ["rev-parse", "--abbrev-ref", "--symbolic-full-name", "@{upstream}"],
63                cwd=path,
64                check=False,
65            )
66            if result.returncode != 0:
67                return (0, 0)
68            remote_branch = result.stdout.strip()
69
70        # Get ahead/behind counts
71        result = run_git(
72            ["rev-list", "--left-right", "--count", f"HEAD...{remote_branch}"],
73            cwd=path,
74            check=False,
75        )
76        if result.returncode != 0:
77            return (0, 0)
78
79        parts = result.stdout.strip().split()
80        if len(parts) == 2:
81            return (int(parts[0]), int(parts[1]))
82        return (0, 0)
83    except Exception:
84        return (0, 0)

Args: path: Repository path (defaults to current directory) remote_branch: Remote branch to compare (e.g., "origin/main") If None, uses tracking branch

Returns: Tuple of (ahead, behind) counts

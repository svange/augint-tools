20class HighlightBar(Static): 

21 """Three-line at-a-glance summary bar. 

22 

23 Line 1: worst-repo + totals + per-org health fractions. 

24 Line 2: refresh status + GitHub rate limits + network ping. 

25 Line 3: spacer that the status bar docks into. 

26 """

41 def rerender(self) -> None: 

42 state = self._state 

43 if state is None or not state.healths: 

44 if state is not None and state.is_refreshing: 

45 top = Text("loading repository data...", style="bold yellow") 

46 else: 

47 top = Text("no data yet. press r to refresh.", style="dim") 

48 self.update(Text("\n").join([top, self._infra_line(), Text(" ")])) 

49 return 

50 

51 worst = min(state.healths, key=lambda h: h.score) 

52 visible = visible_healths(state) 

53 total_problems = sum(1 for h in state.healths if h.worst_severity != Severity.OK) 

54 total_issues = sum(h.status.open_issues for h in state.healths) 

55 total_prs = sum(h.status.open_prs for h in state.healths) 

56 visible_problems = sum(1 for h in visible if h.worst_severity != Severity.OK) 

57 visible_issues = sum(h.status.open_issues for h in visible) 

58 visible_prs = sum(h.status.open_prs for h in visible) 

59 

60 line1 = Text() 

61 line1.append("worst: ", style="bold") 

62 line1.append(f"{worst.status.name} ({worst.worst_severity.name.lower()})") 

63 line1.append(" problems: ", style="bold") 

64 line1.append(f"{visible_problems}/{total_problems}") 

65 line1.append(" issues: ", style="bold") 

66 line1.append(f"{visible_issues}/{total_issues}") 

67 line1.append(" prs: ", style="bold") 

68 line1.append(f"{visible_prs}/{total_prs}") 

69 

70 # Per-org health score fractions. 

71 org_passed: dict[str, int] = defaultdict(int) 

72 org_total: dict[str, int] = defaultdict(int) 

73 for h in state.healths: 

74 org = owner_of(h.status.full_name) 

75 org_passed[org] += h.passed_checks 

76 org_total[org] += h.total_checks 

77 for owner in sorted(org_passed): 

78 p, t = org_passed[owner], org_total[owner] 

79 line1.append(f" {owner}: ", style="bold") 

80 ratio = p / t if t > 0 else 1.0 

81 color = "green" if ratio >= 1.0 else "yellow" if ratio > 0.5 else "red" 

82 line1.append(f"{p}/{t}", style=color) 

83 

84 self.update(Text("\n").join([line1, self._infra_line(), Text(" ")]))

86 def _infra_line(self) -> Text: 

87 """Build the second-line infrastructure status. 

88 

89 Combines refresh countdown, GitHub API rate limits (with explicit 

90 "resets" direction), and network ping on a single compact line. 

91 Styled bold cyan so the user's eye lands on it. 

92 """ 

93 state = self._state 

94 if state is None: 

95 return Text(" ") 

96 

97 line = Text() 

98 now = datetime.now(UTC) 

99 

100 # -- refresh section -- 

101 self._append_refresh(line, state, now) 

102 

103 # -- rate limits section -- 

104 line.append(" ", style="dim") 

105 self._append_rate_limits(line, state, now) 

106 

107 # -- network ping section -- 

108 self._append_ping(line, state) 

109 

110 return line

112 def _append_refresh(self, line: Text, state: AppState, now: datetime) -> None: 

113 """Append compact refresh status: last / next / interval.""" 

114 if state.is_refreshing and state.last_refresh_at is None: 

115 line.append("loading...", style="bold yellow") 

116 return 

117 

118 if state.last_refresh_at is not None: 

119 age = max(0, int((now - state.last_refresh_at).total_seconds())) 

120 line.append(f"last: {_format_age(age)} ago", style="bold cyan") 

121 else: 

122 line.append("last: --", style="bold cyan") 

123 

124 if state.is_refreshing: 

125 line.append(" refreshing...", style="bold yellow") 

126 return 

127 

128 if state.next_refresh_at is not None: 

129 remaining = max(0, int((state.next_refresh_at - now).total_seconds())) 

130 next_label = "now" if remaining == 0 else _format_countdown(remaining) 

131 line.append(f" next: {next_label}", style="bold cyan") 

132 

133 if self._refresh_seconds > 0: 

134 m, s = divmod(self._refresh_seconds, 60) 

135 iv = f"{m}m" if s == 0 else f"{m}m{s:02d}s" if m else f"{s}s" 

136 line.append(f" ({iv})", style="dim cyan") 

137 

138 def _append_rate_limits(self, line: Text, state: AppState, now: datetime) -> None: 

139 """Append GQL + REST rate limits with explicit 'resets' direction.""" 

140 for label, remaining, limit, reset_at in [ 

141 ("GQL", state.gql_remaining, state.gql_limit, state.gql_reset_at), 

142 ("REST", state.rest_remaining, state.rest_limit, state.rest_reset_at), 

143 ]: 

144 used = limit - remaining 

145 ratio = remaining / limit if limit > 0 else 1.0 

146 color = "green" if ratio > 0.5 else "yellow" if ratio > 0.2 else "red" 

147 line.append(f"{label}: ", style="bold") 

148 line.append(f"{used}/{limit}", style=color) 

149 if reset_at is not None: 

150 secs = max(0, int((reset_at - now).total_seconds())) 

151 mins, s = divmod(secs, 60) 

152 reset_label = f"{mins}m{s:02d}s" if mins else f"{s}s" 

153 line.append(f" resets {reset_label}", style="dim") 

154 line.append(" ") 

155 

156 def _append_ping(self, line: Text, state: AppState) -> None: 

157 """Append network ping latency or offline indicator.""" 

158 ping = state.ping_result 

159 if ping is None: 

160 return 

161 if not ping.connected: 

162 dns_fail = sum(1 for d in state.dns_results if not d.resolved) 

163 label = "OFFLINE" 

164 if dns_fail: 

165 label += f" ({dns_fail} DNS)" 

166 line.append(label, style="bold red") 

167 return 

168 if ping.latency_ms is not None: 

169 latency = ping.latency_ms 

170 style = "green" if latency < 50 else "yellow" if latency < 150 else "red" 

171 line.append(f"ping: {latency:.0f}ms", style=style)

243    force_secret: frozenset[str] | set[str] | None = None,
244    comment_hint: str | None = None,

246    """Classify a single key/value pair.
247
248    Parameters
249    ----------
250    key:
251        Environment variable name.
252    value:
253        Environment variable value.
254    force_var:
255        Set of key names that must be classified as variables.
256    force_secret:
257        Set of key names that must be classified as secrets.
258    comment_hint:
259        ``"var"`` or ``"secret"`` parsed from an inline .env comment.
260    """
261    # --- Skip-list check (unchanged) ---
262    if key in KEY_SKIP_PREFIXES:
263        logger.debug(f"[classify] {key} -> SKIP (skip-list key)")
264        return ClassificationResult(key, value, Classification.SKIP, ["skip-list key"])
265
266    if not value:
267        logger.debug(f"[classify] {key} -> SKIP (empty value)")
268        return ClassificationResult(key, value, Classification.SKIP, ["empty value"])
269
270    # --- Explicit overrides (highest priority) ---
271    _force_var = force_var or set()
272    _force_secret = force_secret or set()
273
274    if key in _force_secret:
275        logger.info(f"[classify] {key} -> SECRET (--force-secret override)")
276        return ClassificationResult(key, value, Classification.SECRET, ["--force-secret override"])
277    if key in _force_var:
278        logger.info(f"[classify] {key} -> VARIABLE (--force-var override)")
279        return ClassificationResult(key, value, Classification.VARIABLE, ["--force-var override"])
280
281    # --- Comment hint overrides (second priority) ---
282    if comment_hint == "secret":
283        logger.info(f"[classify] {key} -> SECRET (inline comment hint @secret)")
284        return ClassificationResult(
285            key, value, Classification.SECRET, ["inline comment hint @secret"]
286        )
287    if comment_hint == "var":
288        logger.info(f"[classify] {key} -> VARIABLE (inline comment hint @var)")
289        return ClassificationResult(
290            key, value, Classification.VARIABLE, ["inline comment hint @var"]
291        )
292
293    # --- Always-var allowlist: known infrastructure identifier keys ---
294    # Wins over every heuristic. Still loses to --force-secret and @secret so the
295    # user can override from the call site or the .env itself.
296    if key in KEY_ALWAYS_VAR:
297        logger.info(f"[classify] {key} -> VARIABLE (always-var list)")
298        return ClassificationResult(key, value, Classification.VARIABLE, ["always-var list"])
299
300    # --- Heuristic classification ---
301    reasons: list[str] = []
302
303    # Determine whether the key has an infrastructure suffix (e.g. _role, _bucket, _url).
304    # When it does, weak secret signals (key-keyword, entropy) are suppressed.
305    infra_reason = _is_infra_key(key)
306
307    # Check key-name keywords (potential secret signal)
308    lower_key = key.casefold()
309    for kw in KEY_SECRET_KEYWORDS:
310        if kw in lower_key:
311            if infra_reason:
312                logger.debug(f"[classify] {key}: keyword '{kw}' suppressed by {infra_reason}")
313            else:
314                reasons.append(f"key contains '{kw}'")
315            break
316
317    # Check value prefixes for known secret formats (strong signal)
318    for prefix in VALUE_SECRET_PREFIXES:
319        if value.startswith(prefix):
320            reasons.append(f"value prefix '{prefix}'")
321            break
322
323    # Check structural patterns (strong signals)
324    if _JWT_RE.match(value):
325        reasons.append("JWT pattern")
326    elif _HEX_RE.match(value):
327        reasons.append("long hex string")
328    elif _BASE64_RE.match(value) and len(value) >= 24:
329        reasons.append("base64 blob")
330
331    # Check entropy (weak signal -- suppressed when key has infra suffix)
332    if len(value) >= ENTROPY_MIN_LENGTH:
333        entropy = _shannon_entropy(value)
334        if entropy >= ENTROPY_THRESHOLD:
335            if infra_reason:
336                logger.debug(
337                    f"[classify] {key}: high entropy ({entropy:.2f}) suppressed by {infra_reason}"
338                )
339            else:
340                reasons.append(f"high entropy ({entropy:.2f})")
341
342    # If heuristic flagged it as secret, check safe-value patterns as a final gate.
343    # Safe values (ARNs, URLs) suppress weak secret signals (key-keyword, entropy)
344    # but NOT strong signals (value-prefix, JWT, hex, base64).
345    if reasons:
346        safe_reason = _is_safe_value(value)
347        if safe_reason:
348            strong_signals = {
349                r
350                for r in reasons
351                if r.startswith("value prefix")
352                or r.startswith("JWT")
353                or r.startswith("long hex")
354                or r.startswith("base64")
355            }
356            if not strong_signals:
357                logger.debug(
358                    f"[classify] {key}: secret reasons {reasons} "
359                    f"suppressed by safe value pattern ({safe_reason})"
360                )
361                reasons.clear()
362                reasons.append(f"safe value: {safe_reason}")
363
364    # Secret iff we have reasons AND none are safe-value overrides
365    has_safe = any(r.startswith("safe value:") for r in reasons)
366    if reasons and not has_safe:
367        classification = Classification.SECRET
368    else:
369        classification = Classification.VARIABLE
370
371    logger.debug(f"[classify] {key} -> {classification.value} (reasons: {reasons})")
372    return ClassificationResult(key, value, classification, reasons)

51def resolve_token(filename: str = ".env", auth_source: str = "auto") -> str:
52    """Return a GitHub token from the configured auth source.
53
54    Raises RuntimeError if no token can be found.
55    """
56    dotenv_token = _load_dotenv_values(filename).get("GH_TOKEN", "").strip()
57
58    if auth_source == "dotenv":
59        if dotenv_token:
60            logger.debug("Using GitHub token from GH_TOKEN in .env (--env-auth).")
61            return dotenv_token
62        raise RuntimeError(
63            "No GitHub token found in .env. Remove --env-auth or add GH_TOKEN to .env."
64        )
65
66    if auth_source != "auto":
67        raise ValueError(f"Unsupported auth_source '{auth_source}'.")
68
69    token = os.environ.get("GH_TOKEN", "").strip()
70    if token:
71        logger.debug("Using GitHub token from GH_TOKEN environment variable.")
72        return token
73
74    gh_token = _get_gh_cli_token()
75    if gh_token:
76        if dotenv_token:
77            logger.debug(
78                "Using GitHub token from gh auth token. Ignoring GH_TOKEN from .env; "
79                "export GH_TOKEN in the current shell to force it."
80            )
81        else:
82            logger.debug("Using GitHub token from gh auth token.")
83        return gh_token
84
85    if dotenv_token:
86        logger.debug("Using GitHub token from GH_TOKEN in .env.")
87        return dotenv_token
88
89    raise RuntimeError(
90        "No GitHub token found. Set GH_TOKEN in .env / environment, "
91        "or authenticate with: gh auth login"

1"""Health check: test-coverage threshold enforcement in CI. 

2 

3Parses the repository's canonical GitHub Actions workflow 

4(``publish.yaml`` for libraries, ``deploy.yaml`` for services) and inspects the ``unit-tests`` job 

5for signals that the coverage gate has been **actively weakened**: 

6 

7- Python: ``pytest --cov-fail-under=N`` with N below the canonical 80% baseline 

8 (see the ai-standardize-pipeline standard). Severity LOW. 

9- Any coverage-related step that ignores failures via ``continue-on-error: true`` 

10 or shell suffixes such as ``|| true``, ``|| :``, ``|| echo ...``. Severity MEDIUM. 

11- Python ``pytest --cov`` with no ``--cov-fail-under`` gate at all. Severity LOW 

12 (equivalent to a 0% threshold). 

13- ``unit-tests`` job present but no coverage command inside it. Severity LOW -- 

14 the repo adopted the standardized job name but the gate is missing, which 

15 is a regression from standard, not non-adoption. 

16 

17**True non-adoption is benign.** Repos with no canonical pipeline at all, or 

18with a pipeline that has no ``unit-tests`` job, return OK severity 

19with an informative summary. This is observational, not a warning -- a repo 

20is not "unhealthy" just because it doesn't follow the standard naming 

21convention for its workflow files. 

22 

23For Node repos using ``vitest``/``jest --coverage`` the threshold lives in 

24``vitest.config.*``/``jest.config.*`` rather than the workflow, so this check 

25is intentionally workflow-only and reports OK on seeing the coverage flag -- 

26config-level parsing is out of scope. 

27 

28Workflow text comes from the batched GraphQL snapshot via 

29``FetchContext.pipeline_text`` -- no per-repo REST call. 

30""" 

31 

32from __future__ import annotations 

33 

34import re 

35from typing import TYPE_CHECKING, Any 

36 

37import yaml 

38 

39from .._models import HealthCheckResult, Severity 

40from .._registry import register 

41 

42if TYPE_CHECKING: 

43 from github.Repository import Repository 

44 

45 from ..._data import RepoStatus 

46 from .. import FetchContext

48# Canonical Python coverage threshold enforced by ai-standardize-pipeline. 

49_STANDARD_THRESHOLD = 80 

50 

51# Regex for extracting ``--cov-fail-under=N`` or ``--cov-fail-under N``. 

52_FAIL_UNDER_RE = re.compile(r"--cov-fail-under[=\s]+(\d+)") 

53 

54# Regex for extracting ``fail_under = N`` from pyproject.toml 

55# [tool.coverage.report] section. 

56_PYPROJECT_FAIL_UNDER_RE = re.compile(r"fail_under\s*=\s*(\d+)") 

57 

58# Matches shell suffixes that swallow a non-zero exit code on the same line. 

59# ``|| true``, ``|| :``, ``|| echo "whatever"``. 

60_IGNORE_SUFFIX_RE = re.compile(r"\|\|\s*(true|:|echo\b)") 

61 

62 

63class CoverageCheck: 

64 """Detect lowered or disabled test-coverage gates in CI.""" 

65 

66 name = "coverage" 

67 description = "Detect lowered or ignored test-coverage thresholds in CI" 

68 

69 def evaluate( 

70 self, 

71 repo: Repository, # noqa: ARG002 

72 status: RepoStatus, 

73 *, 

74 config: dict, # noqa: ARG002 

75 context: FetchContext, 

76 ) -> HealthCheckResult: 

77 default_branch = status.default_branch or "main" 

78 used_path = context.pipeline_path 

79 raw = context.pipeline_text 

80 

81 # Missing / non-standard pipeline.yaml is benign -- it just means the 

82 # repo hasn't adopted the ai-standardize-pipeline convention. Surface 

83 # as OK (visible in the detail drawer) but never drives card coloring 

84 # because this is a "standards adoption" observation, not a health 

85 # warning. 

86 if raw is None or used_path is None: 

87 return HealthCheckResult( 

88 check_name=self.name, 

89 severity=Severity.OK, 

90 summary="not standardized (no canonical pipeline)", 

91 ) 

92 

93 workflow_link = f"https://github.com/{status.full_name}/blob/{default_branch}/{used_path}" 

94 

95 try: 

96 workflow = yaml.safe_load(raw) 

97 except yaml.YAMLError: 

98 return HealthCheckResult( 

99 check_name=self.name, 

100 severity=Severity.MEDIUM, 

101 summary="pipeline parse error", 

102 link=workflow_link, 

103 ) 

104 

105 if not isinstance(workflow, dict): 

106 return HealthCheckResult( 

107 check_name=self.name, 

108 severity=Severity.MEDIUM, 

109 summary="pipeline not a mapping", 

110 link=workflow_link, 

111 ) 

112 

113 jobs = workflow.get("jobs") 

114 if not isinstance(jobs, dict): 

115 return HealthCheckResult( 

116 check_name=self.name, 

117 severity=Severity.OK, 

118 summary="pipeline has no jobs", 

119 ) 

120 

121 unit_tests = jobs.get("unit-tests") 

122 if not isinstance(unit_tests, dict): 

123 return HealthCheckResult( 

124 check_name=self.name, 

125 severity=Severity.OK, 

126 summary="not standardized (no unit-tests job)", 

127 ) 

128 

129 steps = unit_tests.get("steps") 

130 if not isinstance(steps, list): 

131 # unit-tests job exists but is empty/malformed -- the standard was 

132 # adopted in name but no gate is actually running. 

133 return HealthCheckResult( 

134 check_name=self.name, 

135 severity=Severity.LOW, 

136 summary="unit-tests job has no steps", 

137 link=workflow_link, 

138 ) 

139 

140 return self._evaluate_steps(steps, workflow_link, context.pyproject_text) 

141 

142 def _evaluate_steps( 

143 self, 

144 steps: list[Any], 

145 workflow_link: str, 

146 pyproject_text: str | None = None, 

147 ) -> HealthCheckResult: 

148 """Walk unit-tests steps and classify the coverage posture.""" 

149 coverage_seen = False 

150 coverage_ignored = False 

151 lowered_threshold: int | None = None 

152 has_python_cov_flag = False 

153 has_python_fail_under = False 

154 has_node_coverage = False 

155 

156 for step in steps: 

157 if not isinstance(step, dict): 

158 continue 

159 run = step.get("run") 

160 if not isinstance(run, str): 

161 continue 

162 

163 is_python = "pytest" in run and "--cov" in run 

164 is_node = ("vitest" in run or "jest" in run) and "--coverage" in run 

165 if not (is_python or is_node): 

166 continue 

167 

168 coverage_seen = True 

169 continue_on_error = bool(step.get("continue-on-error")) 

170 has_ignore_suffix = bool(_IGNORE_SUFFIX_RE.search(run)) 

171 if continue_on_error or has_ignore_suffix: 

172 coverage_ignored = True 

173 

174 if is_python: 

175 has_python_cov_flag = True 

176 match = _FAIL_UNDER_RE.search(run) 

177 if match: 

178 has_python_fail_under = True 

179 threshold = int(match.group(1)) 

180 # Track the lowest threshold seen across all matching steps. 

181 if lowered_threshold is None or threshold < lowered_threshold: 

182 lowered_threshold = threshold 

183 

184 if is_node: 

185 has_node_coverage = True 

186 

187 if not coverage_seen: 

188 # unit-tests job exists but runs no coverage command. Either a 

189 # wrapper hides the real invocation (make/tox/nox/script) or the 

190 # gate was stripped. Either way, from the dashboard's vantage 

191 # point the coverage posture is unverifiable -- flag as LOW so 

192 # the repo surfaces for inspection. 

193 return HealthCheckResult( 

194 check_name=self.name, 

195 severity=Severity.LOW, 

196 summary="unit-tests job has no coverage command", 

197 link=workflow_link, 

198 ) 

199 

200 # Ignoring failures is worse than lowering the bar: still run tests, 

201 # but a red coverage report has no consequence. 

202 if coverage_ignored: 

203 return HealthCheckResult( 

204 check_name=self.name, 

205 severity=Severity.MEDIUM, 

206 summary="coverage failures ignored", 

207 link=workflow_link, 

208 ) 

209 

210 if has_python_fail_under: 

211 assert lowered_threshold is not None 

212 if lowered_threshold < _STANDARD_THRESHOLD: 

213 return HealthCheckResult( 

214 check_name=self.name, 

215 severity=Severity.LOW, 

216 summary=( 

217 f"({lowered_threshold}%/{_STANDARD_THRESHOLD}%) " 

218 f"code coverage reduced from standard" 

219 ), 

220 link=workflow_link, 

221 ) 

222 return HealthCheckResult( 

223 check_name=self.name, 

224 severity=Severity.OK, 

225 summary=f"coverage threshold {lowered_threshold}%", 

226 ) 

227 

228 if has_python_cov_flag: 

229 # No --cov-fail-under on the CLI. The canonical standard places 

230 # the threshold in pyproject.toml [tool.coverage.report] fail_under. 

231 # This check only verifies enforcement EXISTS and isn't bypassed. 

232 # Whether the value meets the standard is coverage.python_threshold_in_config's job. 

233 if pyproject_text and _PYPROJECT_FAIL_UNDER_RE.search(pyproject_text): 

234 return HealthCheckResult( 

235 check_name=self.name, 

236 severity=Severity.OK, 

237 summary="coverage threshold enforced via pyproject.toml", 

238 ) 

239 # No threshold in CLI or config -- CI won't fail on regressions. 

240 return HealthCheckResult( 

241 check_name=self.name, 

242 severity=Severity.LOW, 

243 summary="coverage reported but no fail-under threshold", 

244 link=workflow_link, 

245 ) 

246 

247 if has_node_coverage: 

248 # Workflow-only scope: threshold lives in vitest/jest config files 

249 # and is out of scope for this check. Trust the flag's presence. 

250 return HealthCheckResult( 

251 check_name=self.name, 

252 severity=Severity.OK, 

253 summary="coverage enforced (threshold in config)", 

254 ) 

255 

256 # Defensive fallback -- should be unreachable given the checks above. 

257 return HealthCheckResult( 

258 check_name=self.name, 

259 severity=Severity.OK, 

260 summary="coverage signal unclear", 

261 ) 

262 

263 

264register(CoverageCheck())

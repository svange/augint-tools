1"""Health check that runs the YAML compliance engine. 

2 

3This one check is intentionally different from the others: it returns a 

4**list** of ``HealthCheckResult`` objects, one per rule declared in 

5``standards.yaml``. The runner in ``health/__init__.py`` flattens the list so 

6every rule surfaces as its own finding on the repo card. 

7 

8The engine fetches ``standards.yaml`` from ``augmenting-integrations/ai-cc-tools`` 

9using the same auth token the dashboard's GraphQL queries use. Cache TTL is 

10one hour so edits propagate naturally without thrashing the GitHub API. 

11 

12Results are cached per repo by ``(commit_sha, rulesets_fingerprint)``. When 

13neither the code nor the rulesets have changed, the cached results are 

14returned without re-evaluation. A ruleset change (detected via the REST 

15fetcher's ``updated_at`` tracking) invalidates the cache for that repo. 

16 

17Runtime options (standards URL override, handler registry) come from 

18``config["standards_engine"]`` passed down from ``cmd.py``. 

19""" 

20 

21from __future__ import annotations 

22 

23import hashlib 

24import json 

25from typing import TYPE_CHECKING 

26 

27from .._engine import EngineOptions, run_engine 

28from .._handlers import all_handlers 

29from .._models import HealthCheckResult, Severity 

30from .._registry import register 

31 

32if TYPE_CHECKING: 

33 from github.Repository import Repository

35 from ..._data import RepoStatus 

36 from .. import FetchContext

39def _rulesets_fingerprint(rulesets: list[dict] | None) -> str: 

40 """Stable string representation of rulesets for cache keying.""" 

41 if not rulesets: 

42 return "" 

43 return json.dumps(rulesets, sort_keys=True, default=str) 

44 

45 

46def _overrides_fingerprint(text: str | None) -> str: 

47 """Short hash of the compliance override file so edits bust the cache.""" 

48 if not text: 

49 return "" 

50 return hashlib.sha256(text.encode("utf-8")).hexdigest()[:16] 

51 

52 

53class YamlEngineCheck: 

54 """Loads standards.yaml and emits one result per declared check. 

55 

56 The engine is entirely offline-friendly: when the standards document can't 

57 be fetched and there's no cached copy, it emits a single informational 

58 result and the dashboard carries on. 

59 

60 Results are cached by ``(commit_sha, rulesets_fingerprint)`` per repo. 

61 """ 

62 

63 name = "standards_engine" 

64 description = "Evaluate the canonical standards.yaml compliance rules" 

65 

66 def __init__(self) -> None: 

67 # Cache: repo_full_name -> (cache_key, results). 

68 # Key is (commit_sha, rulesets_fingerprint, overrides_fingerprint) -- 

69 # any of those three changing re-evaluates this repo's checks. 

70 self._cache: dict[str, tuple[tuple[str | None, str, str], list[HealthCheckResult]]] = {} 

71 

72 def evaluate( 

73 self, 

74 repo: Repository, # noqa: ARG002 -- kept for Protocol signature 

75 status: RepoStatus, 

76 *, 

77 config: dict, 

78 context: FetchContext, 

79 ) -> list[HealthCheckResult]: 

80 engine_cfg = (config or {}).get("standards_engine") or {} 

81 gh = engine_cfg.get("gh") 

82 standards_url = engine_cfg.get("url") 

83 

84 repo_key = getattr(status, "full_name", "") or "" 

85 sha = context.main_head_sha 

86 rulesets_fp = _rulesets_fingerprint(context.rulesets) 

87 overrides_fp = _overrides_fingerprint(context.compliance_overrides_text) 

88 cache_key = (sha, rulesets_fp, overrides_fp) 

89 

90 cached = self._cache.get(repo_key) 

91 if cached is not None and cached[0] == cache_key: 

92 return cached[1] 

93 

94 options = EngineOptions(standards_url=standards_url, handlers=all_handlers()) 

95 

96 tags: set[str] = set() 

97 if getattr(status, "is_workspace", False): 

98 tags.add("workspace") 

99 elif getattr(status, "looks_like_service", False): 

100 tags.add("service") 

101 else: 

102 tags.add("library") 

103 if getattr(status, "is_org", False): 

104 tags.add("org") 

105 

106 default_branch = getattr(status, "default_branch", None) or "main" 

107 try: 

108 results = run_engine(context, options, gh, tags, default_branch) 

109 except Exception as exc: 

110 results = [ 

111 HealthCheckResult( 

112 check_name=self.name, 

113 severity=Severity.MEDIUM, 

114 summary=f"standards engine error: {exc.__class__.__name__}: {exc}", 

115 ) 

116 ] 

117 

118 self._cache[repo_key] = (cache_key, results) 

119 return results 

120 

121 

122register(YamlEngineCheck())
